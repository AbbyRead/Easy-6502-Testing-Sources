		;display one random color every 11 ticks

repeat:
JSR getrand
JSR incout	;$0200 is leftmost pixel of each line
JMP repeat	;current line + 1F or next -1

getrand:
LDA $FE		;X contains random value
AND #$0F
RTS

incout:
STA $0200,Y
CPY FF
INY
BEQ nextline1
RTS

nextline1:
JSR getrand
STA $0300,Y
CPY FF
INY
BNE nextline1

nextline2:
JSR getrand
STA $0400,Y
CPY FF
INY
BNE nextline2

nextline3:
JSR getrand
STA $0500,Y
CPY FF
INY
BNE nextline3
RTS
